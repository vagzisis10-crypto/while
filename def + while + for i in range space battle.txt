def space_battle():
   total_hits = 0
   for cannon in range(1, 3, ):
        print(f"enable of cannon {cannon}")
        shots = 0
        while shots < 2:
            shots = shots + 1
            total_hits = total_hits + 1
            print(f"the cannon {cannon} fired the shots {shots}")
            print(f"Η μάχη τελείωσε! Συνολικά χτυπήματα: {total_hits}")

space_battle()
