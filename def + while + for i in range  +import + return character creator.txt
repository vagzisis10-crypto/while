import random

def character_creator():
    classes = ["Warrior", "Mage", "Archer"]
    
    while True:
        print("1. View all classes")
        print("2. Add a new class")
        print("3. Pick a class and exit")
        
        choice = input("Choose an option(1-3):")
        
        if choice == "1":
            for i in range(len(classes)):
                print(f"classes{i + 1}: {classes[i]}")
        elif choice == "2":
            new_class = input("Give another class")
            classes.append(new_class)
            print(f"{new_class} added to {classes}")
        elif choice == "3":
            suggested_class =random.choice(classes)
            return suggested_class
        else:
            print("Error")
            
suggested_class = character_creator()
print(f"the new class is : {suggested_class}")
